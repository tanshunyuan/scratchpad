Browser client resumes after refresh

Server works. Now make browser remember runId and lastEventId.

Goal

Start button:
  POST /runs
  parse SSE events
  save runId from type:"run"
  save lastEventId from type:"delta"

Page load:
  if runId exists:
    GET /runs/:runId/events?after=lastEventId
    parse replay + live events

Client state

const state = {
  get runId() { return localStorage.getItem("runId"); },
  set runId(value) { localStorage.setItem("runId", value); },
  get lastEventId() { return Number(localStorage.getItem("lastEventId") ?? 0); },
  set lastEventId(value) { localStorage.setItem("lastEventId", String(value)); },
};

Parsing rule

The browser gives chunks, not neat SSE events. Your parser buffers text until it sees a blank line.

let buffer = "";

function parseEvents(text) {
  buffer += text;
  const blocks = buffer.split("\n\n");
  buffer = blocks.pop() ?? "";

  return blocks.map((block) => {
    const event = { id: 0, data: null };

    for (const line of block.split("\n")) {
      if (line.startsWith("id: ")) event.id = Number(line.slice(4));
      if (line.startsWith("data: ")) event.data = JSON.parse(line.slice(6));
    }

    return event;
  });
}

Stream reader shape

async function readStream(response) {
  const reader = response.body.getReader();
  const decoder = new TextDecoder();

  while (true) {
    const { value, done } = await reader.read();
    if (done) break;

    for (const event of parseEvents(decoder.decode(value, { stream: true }))) {
      handleEvent(event);
    }
  }
}

Event handling

function handleEvent(event) {
  if (event.data.type === "run") {
    state.runId = event.data.runId;
    return;
  }

  if (event.data.type === "delta") {
    state.lastEventId = event.id;
    output.textContent += event.data.delta + "\n";
  }
}

Start and resume

start.onclick = async () => {
  localStorage.clear();
  const response = await fetch("/runs", { method: "POST" });
  await readStream(response);
};

window.addEventListener("load", async () => {
  if (!state.runId) return;

  const response = await fetch(
    `/runs/${state.runId}/events?after=${state.lastEventId}`,
  );

  if (response.ok) await readStream(response);
});
